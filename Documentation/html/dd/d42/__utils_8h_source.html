98 inline std::string remove_skip_list ( const std::vector< std::string >& words, const std::string& s ) {
99  std::string _return_value = s;
100 
101  for ( auto& __w : words ) {
102  size_t _pos = 0;
103  std::string _lower = boost::algorithm::to_lower_copy ( _return_value );
104 
105  while ( ( _pos = _lower.find ( __w, _pos ) ) != std::string::npos ) {
106  _return_value.erase ( _pos, __w.size() );
107  _lower.erase ( _pos, __w.size() );
108  }
109  }
110 
111  std::regex brace_re ( "(\\( *\\))" );
112  _return_value = std::regex_replace ( _return_value, brace_re, "" );
113  std::regex brackets_re ( "(\\[ *\\])" );
114  _return_value = std::regex_replace ( _return_value, brackets_re, "" );
115  std::regex underscore_re ( "([_*])" );
116  _return_value = std::regex_replace ( _return_value, underscore_re, " " );
117  std::regex dot_re ( "([\\.*])" );
118  _return_value = std::regex_replace ( _return_value, dot_re, " " );
119  boost::algorithm::trim ( _return_value );
120  return _return_value;
121 }
