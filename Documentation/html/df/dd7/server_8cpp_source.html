48 Server::Server ( const std::string& redis,
49  const short port )
50  : redis_ ( data::make_connection ( redis, port ) ), config_ ( json ( data::config ( redis_ ) ) ) {
51  if ( !sub_.connect( redis, port ) ) {
52  spdlog::get ( LOGGER )->error ( "can not subscribe to redis queue" );
53  } else {
54  sub_.subscribe ( EVENT_SCANNER, [] ( const std::string & topic, const std::string & msg ) {
55  spdlog::get ( LOGGER )->debug ( "EVENT:{}:{}", topic, msg );
56  } );
57  sub_.subscribe ( EVENT_RESCAN, [this] ( const std::string & topic, const std::string & msg ) {
58  spdlog::get ( LOGGER )->debug ( "COMMAND:{}:{}", topic, msg );
59  if( !rescanning_ ) {
60  rescanning_ = true;
61  redis_->publish ( EVENT_SCANNER, EVENT_START );
62  scanner_thread_ = std::make_unique< std::thread >( &Server::rescan_, this, msg == "true", [this]( std::error_code& errc ) {
63  spdlog::get( LOGGER )->debug( "scanner fisnished: {}", errc.message() );
64  redis_->publish ( EVENT_SCANNER, EVENT_END );
65  rescanning_ = false;
66  });
67  scanner_thread_->detach();
68  } else spdlog::get( LOGGER )->debug( "scanner already running." );
69  } );
70  }
71 }
