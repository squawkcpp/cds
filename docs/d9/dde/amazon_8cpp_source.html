98 std::string Amazon::generate_hmac256bit_hash(const char *message, const char *key_buf) {
99  unsigned char* result;
100  unsigned int result_len = 32;
101  HMAC_CTX ctx;
102 
103  result = (unsigned char*) malloc(sizeof(char) * result_len);
104 
105  ENGINE_load_builtin_engines();
106  ENGINE_register_all_complete();
107 
108  HMAC_CTX_init(&ctx);
109  HMAC_Init_ex(&ctx, key_buf, strlen( key_buf )/*16*/, EVP_sha256(), NULL);
110  HMAC_Update(&ctx, (unsigned char *)message, strlen( message ) );
111  HMAC_Final(&ctx, result, &result_len);
112  HMAC_CTX_cleanup(&ctx);
113 
114  return std::string( base64( result, result_len ) );
115 }

117 char * Amazon::base64(const unsigned char *input, int length) {
118  BIO *bmem, *b64;
119  BUF_MEM *bptr;
120 
121  b64 = BIO_new(BIO_f_base64());
122  bmem = BIO_new(BIO_s_mem());
123  b64 = BIO_push(b64, bmem);
124  BIO_write(b64, input, length);
125  int res = BIO_flush(b64);
126  if( res < 1 ) {
127  std::cerr << "error calculating mac." << std::endl;
128  }
129  BIO_get_mem_ptr(b64, &bptr);

131  char *buff = (char *)malloc(bptr->length);
132  memcpy(buff, bptr->data, bptr->length-1);
133  buff[bptr->length-1] = 0;
134 
135  BIO_free_all(b64);
136 
137  return buff;
