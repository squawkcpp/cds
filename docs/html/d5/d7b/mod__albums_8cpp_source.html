34 void ModAlbums::import ( data::redis_ptr redis, const config_ptr config, const std::string& key ) {
35  try {
36  std::map< data::NodeType::Enum, std::vector< data::node_t > > _files;
37  import_files ( redis, key, _files );
38  std::string _title, _album, _artist, _composer, _performer,
39  _comment, _year, _track, _disc, _genre, _publisher;
40  bool _cover_found = false;
41  std::string _last_cover = "";
42 
43  for ( auto& __type : _files ) {
44  if ( __type.first == data::NodeType::audio ) {
45  for ( auto& __file : __type.second ) {
46  _title = __file[param::TITLE];
47  _artist = __file[param::ARTIST];
48  _album = __file[param::ALBUM];
49  _year = __file[param::YEAR];
50  _track = clean_track_number( __file[param::TRACK] );
51  _disc = __file[param::DISC];
52 
53  //Get the track information
54  av::Format _format ( __file[param::PATH] );
55  if ( !!_format ) {
56  spdlog::get ( LOGGER )->warn ( "Can not open audio file:{} ({})",
57  _format.errc().message(),
58  __file[param::PATH] );
59  } else {
60  av::Metadata _metadata = _format.metadata();
61 
62  if ( !_metadata.get ( av::Metadata::TITLE ).empty() )
63  { _title = _metadata.get ( av::Metadata::TITLE ); }
64 
65  if ( !_metadata.get ( av::Metadata::ALBUM ).empty() )
66  { _album = _metadata.get ( av::Metadata::ALBUM ); }
67 
68  if ( !_metadata.get ( av::Metadata::ARTIST ).empty() )
69  { _artist = _metadata.get ( av::Metadata::ARTIST ); }
70 
71  if ( !_metadata.get ( av::Metadata::COMPOSER ).empty() )
72  { _composer = _metadata.get ( av::Metadata::COMPOSER ); }
73 
74  if ( !_metadata.get ( av::Metadata::PERFORMER ).empty() )
75  { _performer = _metadata.get ( av::Metadata::PERFORMER ); }
76 
77  if ( !_metadata.get ( av::Metadata::COMMENT ).empty() )
78  { _comment = _metadata.get ( av::Metadata::COMMENT ); }
79 
80  if ( !_metadata.get ( av::Metadata::YEAR ).empty() )
81  { _year = _metadata.get ( av::Metadata::YEAR ); }
82 
83  if ( !_metadata.get ( av::Metadata::TRACK ).empty() )
84  { _track = _metadata.get ( av::Metadata::TRACK ); }
85 
86  if ( !_metadata.get ( av::Metadata::DISC ).empty() )
87  { _disc = _metadata.get ( av::Metadata::DISC ); }
88 
89  if ( !_metadata.get ( av::Metadata::GENRE ).empty() )
90  { _genre = _metadata.get ( av::Metadata::GENRE ); }
91 
92  if ( !_metadata.get ( av::Metadata::PUBLISHER ).empty() )
93  { _publisher = _metadata.get ( av::Metadata::PUBLISHER ); }
94 
95  auto codec = _format.find_codec ( av::CODEC_TYPE::AUDIO );
96  data::save( redis, data::hash ( __file[param::PATH] ), {
97  { param::PARENT, key },
98  { param::TITLE, _title },
99  { param::ALBUM, _album },
100  { param::ARTIST, _artist },
101  { param::COMPOSER, _composer },
102  { param::PERFORMER, _performer },
103  { param::COMMENT, _comment },
104  { param::YEAR, _year },
105  { param::TRACK, clean_track_number( _track ) },
106  { param::DISC, _disc },
107  { param::GENRE, _genre },
108  { param::PUBLISHER, _publisher },
109  { param::BITRATE, std::to_string ( codec->bitrate() ) },
110  { param::BPS, std::to_string ( codec->bits_per_sample() ) },
111  { param::CHANNELS, std::to_string ( codec->channels() ) },
112  { param::SAMPLERATE, std::to_string ( codec->sample_rate() ) },
113  { param::PLAYTIME, std::to_string ( _format.playtime() ) }
114  });
115 
116  //update relation with track as score
117  data::add_types( redis, key, data::hash ( __file[param::PATH] ), data::time_millis() );
118  data::add_nodes( redis, key, data::NodeType::audio,
119  data::hash ( __file[param::PATH] ),
120  std::stoi( clean_track_number( _track ) ) );
121  }
122  }
123 
124  } else if ( __type.first == data::NodeType::image ) {
125  for ( auto& __file : __type.second ) {
126  utils::Image image_meta_ ( __file[param::PATH] );
127  data::save( redis, data::hash ( __file[param::PATH] ), {
128  { param::PARENT, key },
129  { param::CLASS, data::NodeType::str ( data::NodeType::cover ) },
130  { param::WIDTH, std::to_string ( image_meta_.width() ) },
131  { param::HEIGHT, std::to_string ( image_meta_.height() ) }
132  });
133  auto _filename = filename ( __file[param::PATH], false );
134  boost::to_lower ( _filename );
135 
136  //store cover uri in redis map
137  _last_cover = utils::make_cover_uri ( ECoverSizes::TN, data::hash ( __file[param::PATH] ) );
138  if ( std::find ( album_cover_names.begin(), album_cover_names.end(), _filename ) != album_cover_names.end() ) {
139  _cover_found = true;
140  data::save( redis, key, {{ param::THUMB, _last_cover }} );
141  }
142 
143  //scale image
144  image_meta_.scale ( config->tmp_directory, ECoverSizes::MED, data::hash( __file[param::PATH] ) );
145  image_meta_.scale ( config->tmp_directory, ECoverSizes::TN, data::hash( __file[param::PATH] ) );
146 
147  data::add_types( redis, key, data::hash( __file[param::PATH] ), 0 );
148  data::rem_types( redis, key, data::hash( __file[param::PATH] ) );
149  data::rem_nodes( redis, data::NodeType::image, data::hash( __file[param::PATH] ) );
150  data::add_nodes( redis, key, data::NodeType::cover, data::hash ( __file[param::PATH] ),
151  std::find ( album_cover_names.begin(), album_cover_names.end(), _filename ) != album_cover_names.end() ? 1 : 2 );
152  }
153  } else if ( __type.first == data::NodeType::ebook ) {
154  for ( auto& __file : __type.second ) {
155  //delete ebook from new books.
156  redis->command( {redis::SREM, //TODO eBooks in album are still listed in the ebook list.
157  data::make_key( key::FS, key::NEW, data::NodeType::str( data::NodeType::ebook ) ),
158  data::hash( __file[param::PATH] ) } );
159  }
160  } else { spdlog::get ( LOGGER )->debug ( "OTHER TYPE: {}", data::NodeType::str ( __type.first ) ); }
161  }
162 
163  //when no cover is found try to take another image
164  if( !_cover_found && !_last_cover.empty() )
165  { data::set( redis, key, param::THUMB, _last_cover ); }
166 
167  //store folder as album
168  data::save( redis, key, {
169  { param::CLASS, data::NodeType::str ( data::NodeType::album ) },
170  { param::NAME, _album },
171  { param::CLEAN_STRING, clean_string( _album ) },
172  { param::ARTIST, _artist },
173  { param::PERFORMER, _performer },
174  { param::COMMENT, _comment },
175  { param::YEAR, _year },
176  { param::GENRE, _genre }
177  } );
178  //create genre tag for album
179  data::add_tag( redis, param::GENRE, _genre, data::NodeType::movie, key, 0 );
180  //add album relation
181  data::add_nodes( redis, data::NodeType::album, key, data::time_millis() );
182  import_artist ( redis, key, _artist );
183 
184  } catch ( ... ) {
185  spdlog::get ( LOGGER )->error ( "exception mod albums." );
186  }
187 }

189 void ModAlbums::import_files ( data::redis_ptr redis, const std::string& key, std::map< data::NodeType::Enum, std::vector< data::node_t > >& files ) {
190  data::children( redis, key, 0, -1, "default", "asc", "", [&redis,&key,&files](const std::string& __key ) {
191  data::node_t _file = data::node ( redis, __key );
192  if ( _file[param::CLASS] == data::NodeType::str ( data::NodeType::folder ) ) {
193  import_files ( redis, data::hash( _file[param::PATH] ), files );
194  //remove folder and relations
195  //TODO the nodes to this folders still exist
196  data::rem_types( redis, key, _file[param::PATH] );
197  redis->command( {redis::DEL,
198  data::make_key_list( data::hash( _file[param::PATH] ) ),
199  data::make_key_node( data::hash( _file[param::PATH] ) ) } );
200 
201  } else { files[ data::NodeType::parse ( _file[param::CLASS] ) ].push_back ( _file ); }
202  });
203 }
