137 std::vector < std::map<std::string, std::string > > ModMovies::tmdb_parse ( const std::string& result ) {
138  std::vector < std::map<std::string, std::string > > _movies;
139  rapidjson::Document document;
140  document.Parse ( result.c_str() );
141  const auto& results = document[ "results" ];
142  assert ( results.IsArray() );
143 
144  for ( auto itr = results.Begin(); itr != results.End(); ++itr ) {
145  std::map<std::string, std::string > _movie;
146 
147  for ( auto _obj = itr->MemberBegin(); _obj != itr->MemberEnd(); ++_obj ) {
148  if ( strcmp ( _obj->name.GetString(), "id" ) == 0 && _obj->value.IsInt() ) {
149  _movie["id"] = std::to_string( _obj->value.GetInt() );
150  } else if ( strcmp ( _obj->name.GetString(), param::TITLE.c_str() ) == 0 && _obj->value.IsString() ) {
151  _movie[param::TITLE] = _obj->value.GetString();
152  }
153  }
154 
155  _movies.push_back ( _movie );
156  }
157 
158  return _movies;
159 }
