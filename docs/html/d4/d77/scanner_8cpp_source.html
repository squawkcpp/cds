68 void Scanner::import_directory ( data::redis_ptr redis, magic_t& _magic, const std::string& parent_key, const std::string& path ) {
69  //get files in folder
70  boost::filesystem::path _fs_path ( path );
71  boost::filesystem::directory_iterator end_itr;
72 
73  for ( boost::filesystem::directory_iterator itr ( _fs_path ); itr != end_itr; ++itr ) {
74  const std::string _item_filepath = path + "/" + itr->path().filename().string();
75  if ( boost::filesystem::is_regular_file ( itr->status() ) ) {
76  if( ! Scanner::timestamp( redis, data::hash ( _item_filepath ), boost::filesystem::last_write_time( itr->path() ) ) ) {
77  const char* _mime_type = magic_file ( _magic, _item_filepath.c_str() );
78  data::node_t _node;
79  data::NodeType::Enum _type = FileMetaRegex::parse ( _mime_type, _item_filepath.c_str(), _node );
80  _node[param::CLEAN_STRING] = clean_string( _node[param::NAME] );
81  _node[param::PARENT] = data::hash( parent_key );
82  _node[param::PATH] = _item_filepath;
83  _node[param::CLASS] = data::NodeType::str ( _type );
84  _node[param::EXTENSION] = boost::filesystem::extension ( itr->path() );
85  _node[param::SIZE] = std::to_string ( boost::filesystem::file_size ( itr->path() ) );
86  _node[param::MIME_TYPE] = _mime_type;
87  _node[param::TIMESTAMP] = std::to_string( boost::filesystem::last_write_time( itr->path() ) );
88  data::save( redis, data::hash( _item_filepath ), _node );
89  data::add_types( redis, data::hash( parent_key ), data::hash( _item_filepath ), data::time_millis() );
90  data::add_nodes( redis, data::hash( parent_key ), _type, data::hash( _item_filepath ), data::time_millis() );
91  data::incr_mime ( redis, _mime_type );
92  Scanner::new_item(redis, parent_key, data::hash ( _item_filepath ), _type );
93  }
94  } else if ( boost::filesystem::is_directory ( itr->status() ) ) {
95  if( ! Scanner::timestamp( redis, data::hash ( _item_filepath ), boost::filesystem::last_write_time( itr->path() ) ) ) {
96  data::save( redis, data::hash( _item_filepath ), {
97  { param::NAME, filename ( _item_filepath, false ) },
98  { param::CLEAN_STRING, clean_string( filename ( _item_filepath, false ) ) },
99  { param::PARENT, data::hash( path ) },
100  { param::PATH, _item_filepath },
101  { param::CLASS, data::NodeType::str ( data::NodeType::folder ) },
102  { param::TIMESTAMP, std::to_string( boost::filesystem::last_write_time( itr->path() ) ) }
103  });
104  data::add_types( redis, data::hash( parent_key ), data::hash( _item_filepath ), data::time_millis() );
105  }
106  import_directory ( redis, _magic, _item_filepath, _item_filepath );
107  }
108  }
109 }

147 bool Scanner::timestamp( data::redis_ptr redis, const std::string& key, unsigned long timestamp ) {
148  auto& _exist = redis->commandSync< int > ( { redis::EXISTS, data::make_key_node ( key ) } );
149  if( _exist.ok() && _exist.reply() == 1 ) {
150  auto _timestamp = std::stoul( data::get( redis, key, param::TIMESTAMP ) );
151  if( _exist.ok() && _timestamp == timestamp ) { return true; }
152  }
153  return false;
