84 inline std::string _get_ip() {
85  std::string ip_ = "127.0.0.1";
86  struct ifaddrs* ifAddrStruct = NULL;
87  getifaddrs ( &ifAddrStruct );
88  char addressBuffer[INET_ADDRSTRLEN];
89 
90  for ( struct ifaddrs* ifa = ifAddrStruct; ifa != NULL; ifa = ifa->ifa_next ) {
91  if ( ( ifa->ifa_addr )->sa_family == AF_INET ) {
92  void* tmpAddrPtr = & ( ( struct sockaddr_in* ) ifa->ifa_addr )->sin_addr;
93  inet_ntop ( AF_INET, tmpAddrPtr, addressBuffer, INET_ADDRSTRLEN );
94 
95  if ( strcmp ( ifa->ifa_name, "lo" ) ) {
96  ip_ = addressBuffer;
97  }
98  }
99  }
100  if ( ifAddrStruct != NULL ) {
101  freeifaddrs ( ifAddrStruct );
102  }
103  return ip_;
104 }
