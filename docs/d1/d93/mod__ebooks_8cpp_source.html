39 void ModEbooks::import ( data::redis_ptr redis, const config_ptr config, const std::string& key ) {
40  try {
41  data::node_t _file = data::node ( redis, key );
42  std::string _isbn = parsePdf ( _file[param::PATH] );
43 
44  // we have to wait some time, otherwise amazon blocks the requests.
45  sleep ( SLEEP );
46  auto _res = utils::Amazon::bookByIsbn ( config->amazon_access_key, config->amazon_key, _isbn );
47  data::node_t _book_meta;
48 
49  for ( auto& __j : _res.results ) {
50  if ( _book_meta.empty() || __j[param::NAME] == _file[param::NAME] ) {
51  _book_meta = __j;
52  }
53  }
54 
55  if ( !_book_meta.empty() ) {
56  //save image
57  std::string _cover_path = fmt::format ( "{}/{}.jpg", config->tmp_directory, data::hash ( _book_meta[param::COVER] ) );
58  std::ofstream _ofs ( _cover_path, std::ofstream::out );
59  http::get ( _book_meta[param::COVER], _ofs );
60 
61  utils::Image image_meta_ ( _cover_path );
62  image_meta_.scale ( config->tmp_directory, ECoverSizes::MED, key );
63  image_meta_.scale ( config->tmp_directory, ECoverSizes::TN, key );
64 
65 
66  data::save ( redis, data::hash ( _cover_path ), {
67  { param::PARENT, key },
68  { param::CLASS, data::NodeType::str ( data::NodeType::cover ) },
69  { param::WIDTH, std::to_string ( image_meta_.width() ) },
70  { param::HEIGHT, std::to_string ( image_meta_.height() ) }
71  });
72  data::add_nodes( redis, data::hash ( _book_meta[param::COVER] ), data::NodeType::cover,
73  data::make_key_node ( data::hash ( _cover_path ) ), data::time_millis() );
74 
75  //save ebook
76  data::save ( redis, key, {
77  { param::NAME, _book_meta[param::NAME] },
78  { param::COMMENT, _book_meta[param::COMMENT] },
79  { param::PUBLISHER, _book_meta[param::PUBLISHER] },
80  { param::DATE, _book_meta[param::DATE] },
81  { param::ISBN, _book_meta[param::ISBN] },
82  { param::AUTHOR, _book_meta[param::AUTHOR] },
83  { param::THUMB, fmt::format ( "/img/tn_{}.jpg", data::hash ( _book_meta[param::COVER] ) ) },
84  });
85  }
86  auto _last_write_time = boost::filesystem::last_write_time( data::get( redis, key, param::PATH ) );
87  data::add_nodes( redis, data::NodeType::ebook, key, static_cast< unsigned long>( _last_write_time ) );
88  } catch ( ... ) {
89  spdlog::get ( LOGGER )->error ( "exception mod ebooks." );
90  }

93 const std::string ModEbooks::remove_special_characters ( const std::string& body ) {
94  std::string _result = body;
95  std::replace_if ( _result.begin(), _result.end(),
96  [] ( char c ) {
97  return c == '\n';
98  }, '_' );
99  _result.erase ( std::remove_if ( _result.begin(), _result.end(),
100  [] ( char c ) {
101  return ! ( std::isalnum ( c ) || std::isalpha ( c ) || c == '-' || c == ':' || c == ' ' || c == '_' );
102  } ),
103  _result.end() );
104  return _result;
105 }

107 const std::string ModEbooks::find_isbn ( const std::string& body ) {
108  std::string _body = body;
109  static std::array< std::regex, 2 > _reg_list = std::array< std::regex, 2 > ( {
110  { std::regex ( "([\\d]+[- ·]?[\\d]*[- ·]{1}[\\d]*[- ·]{1}[\\d]*[- ·]{1}[\\dX]*)" ),
111  std::regex ( "([\\d]{10,13})" ) }
112  } );
113  boost::to_upper ( _body );
114  size_t _isbn_pos = _body.find ( "ISBN" );
115 
116  if ( _isbn_pos != std::string::npos ) {
117  const std::string _snipped = remove_special_characters ( _body.substr ( _isbn_pos, 100 ) );
118 
119  for ( auto& __re : _reg_list ) {
120  std::smatch match;
121 
122  if ( std::regex_search ( _snipped , match, __re ) && match.size() > 1 )
123  { return match.str ( 1 ); }
124  }
125  }
126 
127  return "";
128 }

133 std::string ModEbooks::parsePdf ( const std::string& filename ) {
134  poppler::document* doc = poppler::document::load_from_file ( filename );
135 
136  if ( doc == nullptr )
137  { return ""; }
138 
139  for ( size_t i = 0U; i < static_cast< size_t > ( doc->pages() ); i++ ) {
140  poppler::page* p = doc->create_page ( static_cast< int >( i ) );
141  std::string _page_string = p->text().to_latin1();
142  std::string _isbn = find_isbn ( _page_string );
143 
144  if ( ! _isbn.empty() ) {
145  return _isbn;
146  }
147  }
148 
149  return "";
150 }
